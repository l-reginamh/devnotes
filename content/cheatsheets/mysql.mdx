```sql title="SELECT"
    SELECT [DISTINCT | TOP n] <ColumnName, ColumnName, ...>
    FROM <TableName.a, TableName.b>
    [INNER JOIN | LEFT JOIN | RIGHT JOIN | FULL OUTER JOIN <TableName.c> ON a.ColumnName=b.ColumnName]
    WHERE <conditions, ...>
    GROUP BY <ColumnName, ...>
    HAVING <conditions, ...>
    ORDER BY <ColumnName, ...>
```
