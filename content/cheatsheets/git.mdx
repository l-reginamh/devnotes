
```bash title="--amend"
git status
git add modified-file.txt
git commit --amend -m "write the new commit message here."
git push origin branch-name -f
```

```bash title="--amend --no-edit"
git status
git add modified-file.txt
git commit --amend --no-edit
git push origin branch-name -f
```
